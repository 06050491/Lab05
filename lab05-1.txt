#include<Keypad.h>
const byte ROWS=4;
const byte COLS=4;
char keys[ROWS][COLS]={{'7','4','1','0'},{'8','5','2','A'},{'9','6','3','B'},{'C','D','E','F'}};
byte rowPins[ROWS]={8,9,10,11};
byte colPins[COLS]={12,13,14,15};
Keypad keypad = Keypad(makeKeymap(keys),rowPins,colPins,ROWS,COLS);
void setup() {
  Serial.begin(9600);

}

void loop() {
  char key =keypad.getKey();
  if(key!=NO_KEY){
      Serial.println(key);
  }

}
